<!-- 复杂数组 -->
求arr中children长度的最大值
let arr = [
    {label: 'aa', children:[{label: 'aaa',}]},
    {label: 'bb', children:[{label: 'bbb',},{label: 'bbbb',}]},
    {label: 'cc', children:[]},
]
let max = Math.max(...arr.map((item)=>{
    return item.children.length
}))
console.log(max)
